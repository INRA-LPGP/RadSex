51 void Map::process_marker(Marker& marker) {
52 
53  this->best_alignments[0] = 0;
54  this->best_alignments[1] = 0;
55  this->best_alignments[2] = 0;
56 
57  if (marker.n_individuals >= this->min_individuals) {
58 
59  this->alignment_results = mem_align1(this->bwa_mem_options, this->index->bwt, this->index->bns, this->index->pac, marker.sequence.size(), marker.sequence.c_str()); // Align marker to the reference
60 
61  for (uint j = 0; j < this->alignment_results.n; ++j) { // Loop through alignments to find best scoring one
62 
63  if (this->alignment_results.a[j].score > this->best_alignments[1]) { // Current alignment has best score so far, store its properties
64 
65  this->best_alignments[0] = static_cast<int>(j);
66  this->best_alignments[1] = this->alignment_results.a[j].score;
67  this->best_alignments[2] = 1;
68 
69  } else if (this->alignment_results.a[j].score == this->best_alignments[1]) { // Current alignment is as good as current best alignment, increase count
70 
71  ++this->best_alignments[2];
72 
73  }
74 
75  }
76 
77  // Get best alignment info
78  this->best_alignment = mem_reg2aln(this->bwa_mem_options, this->index->bns, this->index->pac, marker.sequence.size(), marker.sequence.c_str(), &this->alignment_results.a[this->best_alignments[0]]);
79 
80  // Retain aligned markers with unique best alignment and mapq >= minimum quality (rid < 0 means not aligned)
81  if (this->best_alignment.rid >= 0 and this->best_alignments[2] == 1 and this->best_alignment.mapq >= this->parameters.map_min_quality) {
82 
83  // Compute group bias. Assumption: no empty group (checked in map function)
84  this->aligned_marker.bias = float(marker.group_counts.at(this->parameters.group1)) / float(this->popmap.get_count(this->parameters.group1)) - float(marker.group_counts.at(this->parameters.group2)) / float(this->popmap.get_count(this->parameters.group2));
85  this->aligned_marker.p = get_p_association(marker.group_counts.at(this->parameters.group1), marker.group_counts.at(this->parameters.group2), this->popmap.get_count(this->parameters.group1), this->popmap.get_count(this->parameters.group2));
86  this->aligned_marker.id = marker.id;
87  this->aligned_marker.contig = this->index->bns->anns[this->best_alignment.rid].name;
88  this->aligned_marker.position = this->best_alignment.pos;
89  this->results.aligned_markers.push_back(this->aligned_marker);
90 
91  }
92 
93  free(this->best_alignment.cigar); // Deallocate cigar string for best hit
94  free(this->alignment_results.a); // Deallocate the hit list
95 
96  }
97 
98  if (marker.n_individuals > 0) ++this->results.n_markers;
99 
100 }

130  std::ifstream genome_file = open_input(this->parameters.genome_file_path);

132  std::string line = "", contig_name = "";

136

140

143  contig_name = contig_name.substr(1, contig_name.size());
144  contig_length = 0;

147

149 
150  }
151  }
152 
153  this->contig_lengths[contig_name] = contig_length; // Last contig
154  genome_file.close();

162 void Map::load_bwa_index() {
163 
164  std::ifstream bwa_index_temp;
165  bool indexed = true;
166  std::string extensions[5] = {"amb", "ann", "bwt", "pac", "sa"}; // BWA index file extensions
167 
168  // Check whether all index files exist
169  for (auto i=0; i<5; ++i) {
170 
171  bwa_index_temp.open(this->parameters.genome_file_path + "." + extensions[i]);
172  if (bwa_index_temp.is_open()) bwa_index_temp.close(); else indexed = false;
173 
174  }
175 
176  // If index does not exist, build it
177  if (not indexed) {
178 
179  log("BWA index file not found for the reference, creating bwa index file");
180  bwa_idx_build(this->parameters.genome_file_path.c_str(), this->parameters.genome_file_path.c_str(), 0, 10000000); // Arguments: genome file, prefix, algorithm (default 0), block size (default 10000000)
181 
182  }
183 
184  // Load index
185  this->index = bwa_idx_load(this->parameters.genome_file_path.c_str(), BWA_IDX_ALL);
186 
187  if (this->index == nullptr) { // Check that index was loaded properly
188 
189  log("Failed to load index for genome file <" + this->parameters.genome_file_path + ">", LOG_ERROR);
190  exit(1);
191 
192  }

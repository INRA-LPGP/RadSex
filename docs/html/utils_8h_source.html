246 inline void log_progress_bar(uint64_t& n_processed_markers, const uint32_t marker_processed_tick, char symbol = '#', uint16_t ticks = 50) {
247 
248  uint tick_size = 100 / ticks; // Compute the number of markers in a tick
249 
250  if (n_processed_markers % (tick_size * marker_processed_tick) != 0) return; // Not an update step, stop there
251 
252  uint16_t progress = static_cast<uint16_t>(n_processed_markers / (tick_size * marker_processed_tick)); // Compute current progress
253 
254  std::string bar = "Progress: [";
255 
256  for(uint i=0; i < ticks; ++i) i < progress ? bar += symbol : bar += '.'; // Draw progress bar
257 
258  bar += "] - " + std::to_string(progress * tick_size) + "% (" + std::to_string(n_processed_markers) + " markers)";
259 
260  progress < ticks ? bar += '\r' : bar += '\n';
261 
262  log(bar, LOG_INFO, false);
263 
264 }

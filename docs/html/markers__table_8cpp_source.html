70 void MarkersTable::start_parser(bool store_sequence, bool compute_groups) {
71

73

76 
77  }
78 
79  std::ifstream input_file = open_input(this->parameters->markers_table_path);
80 
81  if (compute_groups) this->get_group_columns();
82 
83  // Define variables used to read the file

85  uint k = 0, field_n = 0, marker_n = 0;
86  Marker marker(this->header.n_individuals);

88  uint tmp_queue_real_size = 0;
89  uint marker_queue_size = 0;
90  std::string temp = "#";
91 
92  while (temp[0] == '#') std::getline(input_file, temp); // Read comment lines and header line
93 
94  temp = "";
95 
96  do {
97 
98  // Read a chunk of size given by the buffer
99  input_file.read(buffer, sizeof(buffer));
100  k = static_cast<uint>(input_file.gcount());
101 
102  for (uint i=0; i<k; ++i) {
103 
104  // Read the buffer character by character
105  switch(buffer[i]) {
106 
107  case '\t': // New field
108 
109  switch (field_n) {
110 
111  case 0:
112  if (store_sequence) marker.id = temp;
113  break;
114 
115  case 1:
116  if (store_sequence) marker.sequence = temp;
117  break;
118 
119  default:
120  marker.individual_depths[field_n - 2] = static_cast<uint16_t>(fast_stoi(temp.c_str()));
121  if (marker.individual_depths[field_n - 2] >= this->parameters->min_depth) {
122  if (compute_groups) ++marker.group_counts[this->groups[field_n]];
123  ++marker.n_individuals;
124  }
125  break;
126 
127  }
128 
129  temp = "";
130  ++field_n;
131  break;
132 
133  case '\n': // New line (also a new field)
134 
135  marker.individual_depths[field_n - 2] = static_cast<uint16_t>(fast_stoi(temp.c_str()));
136  if (marker.individual_depths[field_n - 2] >= this->parameters->min_depth) {
137  if (compute_groups) ++marker.group_counts[this->groups[field_n]];
138  ++marker.n_individuals;
139  }
140 
141  // Add marker to the queue
142  tmp_queue[marker_n % TMP_QUEUE_SIZE] = marker;
143  ++tmp_queue_real_size;
144  ++marker_n;
145 
146  // Merge temporary queue with global queue
147  if (marker_n % TMP_QUEUE_SIZE == 0) {
148 
149  do {
150 
151  marker_queue_size = static_cast<uint>(this->markers_queue.size());
152  if (marker_queue_size > MAX_QUEUE_SIZE) std::this_thread::sleep_for(std::chrono::microseconds(10)); // Wait if queue is too big
153 
154  } while (marker_queue_size > MAX_QUEUE_SIZE);
155 
156  this->mutex.lock();
157 
158  for (auto& tmp_marker: tmp_queue) {
159 
160  this->markers_queue.push(tmp_marker);
161  ++this->n_markers_processed;
162  tmp_marker.reset(store_sequence);
163 
164  }
165 
166  this->mutex.unlock();
167 
168  tmp_queue_real_size = 0;
169 
170  }
171 
172  // Reset variables
173  temp = "";
174  field_n = 0;
175  marker.reset(store_sequence);
176  break;
177 
178  default:
179 
180  temp += buffer[i];
181  break;
182 
183  }
184 
185  }
186 
187  } while (input_file);
188 
189  // Last batch: real number of markers to push is given by tmp_queue_real_size
190  this->mutex.lock();
191 
192  for (uint i=0; i<tmp_queue_real_size; ++i) {
193 
194  this->markers_queue.push(tmp_queue[i]);
195  ++this->n_markers_processed;
196 
197  }
198 
199  this->mutex.unlock();
200 
201  input_file.close();
202 
203  this->parsing_ended = true; // Shared flag indicating that the parsing is finished
204
